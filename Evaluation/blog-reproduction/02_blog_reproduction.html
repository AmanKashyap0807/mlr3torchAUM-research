3. Custom Stratified Batch Sampler

library(data.table)
batch_sampler_stratified <- function(min_samples_per_stratum, shuffle=TRUE){
  torch::sampler(
    "StratifiedSampler",
    initialize = function(data_source) {
      self$data_source <- data_source
      TSK <- data_source$task
      self$stratum <- TSK$col_roles$stratum
      if(length(self$stratum)==0)stop(TSK$id, "task missing stratum column role")
      self$stratum_dt <- data.table(
        TSK$data(cols=self$stratum),
        row.id=1:TSK$nrow)
      self$set_batch_list()
    },
    set_batch_list = function() {
      get_indices <- if(shuffle){
        function(n)torch::as_array(torch::torch_randperm(n))+1L
      }else{
        function(n)1:n
      }
      index_dt <- self$stratum_dt[
        get_indices(.N)
      ][
      , i.in.stratum := 1:.N, by=c(self$stratum)
      ][]
      count_dt <- index_dt[, .(
        max.i=max(i.in.stratum)
      ), by=c(self$stratum)][order(max.i)]
      count_min <- count_dt$max.i[1]
      num_batches <- max(1, count_min %/% min_samples_per_stratum)
      max_samp <- num_batches * min_samples_per_stratum
      index_dt[
      , n.samp := i.in.stratum/max(i.in.stratum)*max_samp
      , by=c(self$stratum)
      ][
      , batch.i := ceiling(n.samp/min_samples_per_stratum)
      ][]
      save_list$count[[paste("epoch", length(save_list$count)+1)]] <<- dcast(
        index_dt,
        batch.i ~ Class,
        list(length, indices=function(x)paste(x, collapse=",")),
        value.var="row.id"
      )[
      , labels := index_dt[, paste(Class, collapse=""), by=batch.i][order(batch.i), V1]
      ]
      self$batch_list <- split(index_dt$row.id, index_dt$batch.i)
      self$batch_sizes <- sapply(self$batch_list, length)
      self$batch_size_tab <- sort(table(self$batch_sizes))
      self$batch_size <- as.integer(names(self$batch_size_tab)[length(self$batch_size_tab)])
    },
    .iter = function() {
      batch.i <- 0
      function() {
        if (batch.i < length(self$batch_list)) {
          batch.i <<- batch.i + 1L
          indices <- self$batch_list[[batch.i]]
          save_list$indices[[length(save_list$indices)+1]] <<- indices
          if (batch.i == length(self$batch_list)) {
            self$set_batch_list()
          }
          return(indices)
        }
        coro::exhausted()
      }
    },
    .length = function() {
      length(self$batch_list)
    }
  )
}

4. Learner and Loss

save_list <- list()
nn_bce_with_logits_loss_save <- torch::nn_module(
  "nn_print_loss",
  inherit = torch::nn_mse_loss,
  initialize = function() {
    super$initialize()
    self$bce <- torch::nn_bce_with_logits_loss()
  },
  forward = function(input, target) {
    save_list$target[[length(save_list$target)+1]] <<- target
    self$bce(input, target)
  }
)
set.seed(4)
mlp_learner <- mlr3torch::LearnerTorchMLP$new(
  task_type="classif",
  loss=nn_bce_with_logits_loss_save)
mlp_learner$predict_type <- "prob"
mlp_learner$param_set$set_values(
  epochs=1,
  p=0, # dropout probability.
  batch_size=1, # ignored.
  batch_sampler=batch_sampler_stratified(min_samples_per_stratum = 1))

9.3 Exercise: replace torch_randperm with sample

The blog post notes: > “Note that this would not have been possible if we used R’s randomness in batch_sampler_stratified (exercise for the reader: replace torch::torch_randperm with sample and show that you get different weights between repetitions).”

Why this happens: mlr3torch internally calls torch_manual_seed(seed) before each training run, which controls torch’s pseudo-random number generator. However, torch_manual_seed does not seed R’s built-in RNG. If our custom sampler uses R’s sample(), each repetition gets different batch orderings despite the same seed=1, causing gradient descent to follow different optimization paths and produce different final weights.

Below we define a stripped-down version of the sampler (without the save_list instrumentation, as we only need the final weights) that uses sample(n).

batch_sampler_stratified_rsample <- function(min_samples_per_stratum, shuffle=TRUE){
  torch::sampler(
    "StratifiedSamplerRsample",
    initialize = function(data_source) {
      self$data_source <- data_source
      TSK <- data_source$task
      self$stratum <- TSK$col_roles$stratum
      self$stratum_dt <- data.table(
        TSK$data(cols=self$stratum), row.id=1:TSK$nrow)
      self$set_batch_list()
    },
    set_batch_list = function() {
      get_indices <- if(shuffle){
        function(n) sample(n)  # <-- REPLACED: R's sample() instead of torch_randperm()
      }else{
        function(n) 1:n
      }
      index_dt <- self$stratum_dt[get_indices(.N)][
        , i.in.stratum := 1:.N, by=c(self$stratum)][]
      
      count_dt <- index_dt[, .(max.i=max(i.in.stratum)), by=c(self$stratum)][order(max.i)]
      num_batches <- max(1, count_dt$max.i[1] %/% min_samples_per_stratum) 
      max_samp <- num_batches * min_samples_per_stratum
      
      index_dt[, n.samp := i.in.stratum/max(i.in.stratum)*max_samp, by=c(self$stratum)][
        , batch.i := ceiling(n.samp/min_samples_per_stratum)][]
      self$batch_list <- split(index_dt$row.id, index_dt$batch.i)
    },
    .iter = function() {
      batch.i <- 0
      function() {
        if (batch.i < length(self$batch_list)) {
          batch.i <<- batch.i + 1L
          indices <- self$batch_list[[batch.i]]
          if (batch.i == length(self$batch_list)) self$set_batch_list()
          return(indices)
        }
        coro::exhausted()
      }
    },
    .length = function() length(self$batch_list)
  )
}
